##创建对象的三种方式
1.字面量方式
var per1={ name:'卡卡西', age:20, sex:'男', eat:function(){ console.log('吃臭豆腐') }, readBook:function(){ console.log('亲热天堂') } };
2.调用系统的构造函数
var per2 = new Object(); per2.name="大蛇丸"; per2.age="30"; per2.sex="男"; per2.eat=function(){ console.log("吃榴莲") }; per2.play=function(){ console.log("玩蛇") };
//以上两种方式创建的对象都不能确定属于什么类型，例如 var dog = new Object(),狗和人都属于object类型。

`
function Person(name,age){
this.name=name;this.age=age;
this.sayhi=function(){
console.log(“hi”)
}
}

3.设置属性和方法的值
4.把this对象返回 /

##工厂模式创建对象
function createObject(name,age){ var obj=new object(); obj.name=name;obj.age=age; obj.saihi=function(){ console.log('hi') }; return obj } var per1 = new Person("自来也",50) //自定义构造函数 var per2 = createObject("佐助",18); //工厂模式

共同点：都可以创建对象，都可以传入参数
工厂模式：函数名是小写

直接调用函数就可以创建对象
